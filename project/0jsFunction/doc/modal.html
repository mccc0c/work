Modal

弹出框

Options

dialog: '.dialog', //弹出层的名字
bgscroll: false, //optional,背景是否可以滚动
width: null, //optional,宽
height: null, //optional,高
dragable: true, //optional,是否可以拖动
ani: true, //optional,是否有css3动画
bgclose: false, //optional,点击背景关闭
mobileDetect: true, //optional,是否自动判断移动端
myalert: false, //optional,是否只有确定按钮
alertCenter: true, //optional,弹框文字是否居中
hideDuration: 0, //optional,弹框自动消失的时间
top: '50%', //optional,top值
left: '50%', //optional,left值
beforeOpen: function() {}, //optional,弹出前回调
onOpen: function() {}, //optional,弹出后回调
onClose: function() {}, //optional,关闭或取消后回调
onOk: function() {} //optional,确定后回调

Javascript

$('.dopen').modal({
		dialog: '.dialog'
});
