Javascript

var playlist;
$(function() {
	$.ajax({
		url: 'images/songs.json',
		type: 'get',
		dataType: 'json',
		success: function(r) {
			playlist = r;
			$('#result').tmpl('tmpl', playlist); //tmpl: 模板对应的id, playlist: json数据
		}
	})
});

Html

//模板的写法
<script type="text/html" id="tmpl">
<% for ( var i = 0; i < playlist.length; i++ ) { %>
	<li><%=playlist[i].title%><br><span><%=playlist[i].singer%></span></li>
<% } %>
</script>
