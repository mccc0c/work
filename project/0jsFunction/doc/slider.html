Options

mode: 'horizontal',//optional,动画效果，horizontal, vertical, fade
event: 'click',//事件
loop: true,//optional,循环播放
slideMargin: 0,
start: 0,//optional,从第几个开始
captions: false,//optional,标题
ticker: false,//报价机
tickerHover: false,
disableBtn: false,//optional,结束时禁用按钮
pager: true,//optional,圆点
auto: false,//optional,自动播放
pause: 4000,
// carousel
num: 1,//左右滚动显示几个
move: 0,//左右滚动每次移动几个
slideWidth: 0,//optional,单个宽度
// gallery
gallery: false,
galleryMode: 'horizontal',
galleryNum: 5,
galleryMargin: 5,
galleryMove: 1,
// callbacks
onLoad: function() {},
onSlideBefore: function() {},
onSlideAfter: function() {},
onSlideNext: function() {},
onSlidePrev: function() {},
onResize: function() {}

slider.slideTo();//滚动到第几页

slider.reload();//重新渲染
slider.destroy();
slider.update();
slider.page();
slider.startAuto();
slider.stopAuto();
slider.slideNext();
slider.slidePrev();
