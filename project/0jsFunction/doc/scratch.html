Scratch

刮刮卡

Options

image: "images/scratch02.jpg", //optional, false 灰色背景
percent: 0.85, //optional, 剩余百分比算完成
radius: 30 //optional, 半径

Javascript

$('#canvas').scratch({
	image: "images/scratch02.jpg", //false 灰色背景
	percent: 0.85,
	radius: 30
});

$('#canvas2').scratch({
	image: false,
	percent: 0.5,
	radius: 8
});
