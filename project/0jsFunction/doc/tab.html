Tab

Tab切换

Options

event: 'mouseenter',//optional,事件，点击还是hover
auto: true,//optional,自动播放
duration: 3600,//optional,自动播放时间
nav: '.nav',//optional,菜单的class
content: '.content',//optional,内容的class
before: function() {},//optional,开始切换
after: function() {},//optional,切换完成

Javascript

$('.tab').tab();

Html

<div class="tab">
  <div class="nav">
    <ul>
      <li class="current">nav1</li>
      <li>nav2</li>
      <li>nav3</li>
      <li>nav4</li>
      <li>nav5</li>
    </ul>
  </div>
  <div class="content">
    <ul>
      <li style="display:block">content1</li>
      <li>content2</li>
      <li>content3</li>
      <li>content4</li>
      <li>content5</li>
    </ul>
  </div>
</div>

Css

.tab li { list-style: none; }
.tab .nav { height: 30px; }
.tab .nav li { cursor:pointer; width: 50px; height: 30px; float: left; margin-right: 10px; background: #eee}
.tab .nav li:hover, .tab .nav .current { background: #666; color: #fff}
.tab .content>ul>li { display: none}
