Slidetouch

移动端触屏滑动

Options

width: $(document).width(),//optional,宽度
height: $(document).height(),//optional,高度
direction:'horizontal',//optional,方向
nav:false,//optional,圆点导航
onEnd: function() {}//optional,幻灯结束时回调

Methods

$('#slidetouch1').slideTouch.slideIndex()//optional,当前页index

Javascript

var aslide = $('#slidetouch1').slideTouch({
		//'direction':'horizontal',
		//'nav':true,
		'onEnd': function() {
			//console.log($('#slidetouch1').slideTouch.slideIndex());
		}
	});

Html

<div class="slidetouch" id="slidetouch1">
  <div class="slide"><img src="images/01.jpg" />
  <div class="slide"><img src="images/02.jpg" />
  <div class="slide"><img src="images/03.jpg" />
  <div class="slide"><img src="images/04.jpg" />
</div>
