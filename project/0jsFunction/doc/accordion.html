Accordion

经过展开

Options

event: 'mouseenter',//optional,事件，点击还是hover
showfirst: true,//optional, 默认显示第一个
collapse: true,//optional, 自身是否可以开合
animate: true,//optional, slide动画
duration: 300,//optional, slide动画时间
nav: 'dt',//optional, 标题的class
content: 'dd'//optional, 内容的class
active: 'on'//optional, 激活的class

Javascript

$('.accordion').accordion();

Html

<dl class="accordion">
  <dt>title</dt>
  <dd>content</dd>
  <dt>title</dt>
  <dd>content</dd>
  <dt>title</dt>
  <dd>content</dd>
</dl>

Css

.tab li { list-style: none; }
.tab .nav { height: 30px; }
.tab .nav li { cursor:pointer; width: 50px; height: 30px; float: left; margin-right: 10px; background: #eee}
.tab .nav li:hover, .tab .nav .current { background: #666; color: #fff}
.tab .content>ul>li { display: none}
